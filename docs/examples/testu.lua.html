require 'winapi'
local U = winapi.utf8_expand
local UTF8 = winapi.CP_UTF8

winapi.set_encoding(UTF8)

function bytes (txt)
    return (txt:gsub('.',function(c)
        c = c:byte(1)
        local hi = math.floor(c/128) == 1
        if hi then
            return ('[%x]'):format(c)
        else
            return ('%c'):format(c)
        end
    end))
end

function paircount (txt)
    local floor = math.floor
    local k,i = 0,1
    repeat
        if floor(txt:byte(i)/128) == 1 then
            i = i + 1
        end
        k = k + 1
        i = i + 1
    until i > #txt
    return k
end

txt = U '#03BB+#03BC+C*#03C0+#03C3'
print(txt)
print(bytes(txt))
print(paircount(txt))
